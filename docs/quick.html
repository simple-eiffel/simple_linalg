Quick Examples

2D Vector

local
    l_v: VECTOR_2
do
    create l_v.make_from_components (3.0, 4.0)
    Result := l_v.norm  -- Returns 5.0
end

3D Vector with Cross Product

local
    l_a, l_b, l_c: VECTOR_3
do
    create l_a.make_from_components (1.0, 0.0, 0.0)
    create l_b.make_from_components (0.0, 1.0, 0.0)
    l_c := l_a.cross_product (l_b)
    -- l_c = (0.0, 0.0, 1.0)
end

N-Dimensional Vector

local
    l_v: VECTOR_N
do
    create l_v.make_zero (5)
    -- l_v = [0, 0, 0, 0, 0]
    l_v.set_at (1, 2.0)
    Result := l_v.norm (1)  -- L1 norm
end

Matrix Operations

local
    l_m1, l_m2, l_m3: MATRIX
do
    create l_m1.make_identity (3)
    create l_m2.make_identity (3)
    l_m3 := l_m1.multiply (l_m2)
    Result := l_m3.is_identity  -- Returns true
end

Condition Number Check

local
    l_m: MATRIX
    l_x, l_b: VECTOR_N
do
    create l_m.make_identity (3)
    create l_b.make_zero (3)
    if l_m.condition_number <= 1.0e12 then
        l_x := l_m.solve (l_b)
    end
end

Eigenvalue Computation

local
    l_m: MATRIX
    l_lambda: REAL_64
    l_v: VECTOR_N
do
    create l_m.make_identity (3)
    l_lambda := l_m.dominant_eigenvalue
    l_v := l_m.dominant_eigenvector
end

Common Operations

  • Vector dot product: l_a.dot_product (l_b)
  • Vector norm: l_v.norm (L2 by default)
  • Matrix transpose: l_m.transpose
  • Matrix multiplication: l_m1.multiply (l_m2)
  • Linear solve (Ax=b): l_m.solve (l_b)
  • Condition number: l_m.condition_number

Result Types

  • VECTOR_2, VECTOR_3: Fixed-dimension specialized vectors
  • VECTOR_N: N-dimensional generic vector
  • MATRIX: MÃ—N matrix with robust algorithms
  • EIGENVALUE_RESULT: Eigenvalue + eigenvector with convergence info
  • LU_DECOMPOSITION: L and U factors from decomposition
